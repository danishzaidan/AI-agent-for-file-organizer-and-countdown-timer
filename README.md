pastikan ada file toolkit.py dan jalankan di terminal, code dibawah 

# Install dependency
pip install psutil

# File Organizer (Move File)
python toolkit.py organize
python toolkit.py organize /path/to/folder

# System Info
python toolkit.py systeminfo

# Countdown Timer
python toolkit.py timer 60
python toolkit.py timer 300 --message "Coffee break over!"

# Pomodoro Timer
python toolkit.py pomodoro
python toolkit.py pomodoro --work 30 --break 10

# Help
python toolkit.py --help
